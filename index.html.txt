<script>
    // Simple 10-minute countdown logic
    let time = 600; // 10 minutes in seconds
    const minDisplay = document.getElementById('minutes');
    const secDisplay = document.getElementById('seconds');

    function updateTimer() {
        const minutes = Math.floor(time / 60);
        let seconds = time % 60;
        
        seconds = seconds < 10 ? '0' + seconds : seconds;
        
        minDisplay.innerHTML = `${minutes} Min`;
        secDisplay.innerHTML = `${seconds} Sec`;

        if (time > 0) {
            time--;
        } else {
            time = 600; // Reset timer when it reaches zero (typical for landing pages)
        }
    }

    setInterval(updateTimer, 1000);
</script>
